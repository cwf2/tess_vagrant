server {
   listen 80; 
   server_name valery.tesserae.org;
   access_log /var/log/nginx/tesserae.access.log;
   error_log /var/log/nginx/tesserae.error.log;
   root /vagrant/tesserae/html;
   index index.html;

   location /cgi-bin/ {
      root /vagrant/tesserae/;

      location ~* .pl$ {
         try_files $uri =404;
         fastcgi_pass unix:/var/run/fcgiwrap.socket;
         include fastcgi_params;
      }
   }

   location /css/ {
      root /vagrant/tesserae;
   }

   location /images/ {
      root /vagrant/tesserae;
   }
}
